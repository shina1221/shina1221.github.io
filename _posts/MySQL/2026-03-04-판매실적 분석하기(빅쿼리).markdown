```
-- SELECT * FROM `pj-1-489007.part1.sales` LIMIT 1000;

-- 시간의 흐름에 따른 판매 실적을 분석해 미래의 sales와 profit 예측하기  
-- 날짜별 주문 실적을 정리할 수 있는 쿼리  
select(invoicedate) as orderdate,
sum(quantity) as totalquantity,
sum(quantity * unitprice) as totalsales,
sum(quantity * unitprice * 0.2) as totalprofit
from `pj-1-489007.part1.sales` 
group by orderdate
order by orderdate;
```
